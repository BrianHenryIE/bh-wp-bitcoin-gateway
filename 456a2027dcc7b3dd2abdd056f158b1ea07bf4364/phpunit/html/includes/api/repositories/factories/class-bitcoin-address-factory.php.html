176        /** @var mixed|MoneySerializedArray $meta_value */
177        $meta_value = get_post_meta( $post_id, $meta_key, true );
178        if ( empty( $meta_value ) ) {

189            $this->log_meta_value_warning( $post_id, $meta_key, 'Money', $meta_value, '', $exception );

191        }
